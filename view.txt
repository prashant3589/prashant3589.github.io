$org = "$(System.CollectionUri)"
$project = "$(System.TeamProject)"
$runId = "$(Build.BuildId)"
$url = "$org$project/_apis/pipelines/approvals?runId=$runId&api-version=7.1-preview"
$headers = @{
    Authorization = "Bearer $(System.AccessToken)"
}
$response = Invoke-RestMethod -Uri $url -Headers $headers
Write-Host "Approval Response:"
$response | ConvertTo-Json -Depth 50 | Write-Host
$comment = $response.value[0].approvalSteps[0].comment
Write-Host "Change ID entered by approver: $comment"
Write-Host "##vso[task.setvariable variable=CHANGE_ID]$comment"
