using System;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        // Path to the JSON file
        string filePath = "data.json";  // Ensure this file exists in your project directory

        // Step 1: Read JSON content from the file
        string jsonString = File.ReadAllText(filePath);

        // Step 2: Convert JSON string to bytes (UTF-8)
        byte[] bytes = Encoding.UTF8.GetBytes(jsonString);

        // Step 3: Convert bytes to Base64 string
        string base64 = Convert.ToBase64String(bytes);

        // Output the result
        Console.WriteLine("Base64 Encoded JSON:");
        Console.WriteLine(base64);
    }
}
