- powershell: |
    # Download and run Poetry install script
    curl -sSL https://install.python-poetry.org | python -

    # Add Poetry's install path to PATH for current session
    $poetryPath = "$env:APPDATA\Python\Scripts"
    $env:Path += ";$poetryPath"

    # Verify poetry
    poetry --version
  displayName: 'Install Poetry and set PATH'


$runner = Get-ChildItem -Recurse -Filter runner.py | Select-Object -First 1
    Write-Host "Running: $($runner.FullName)"
    poetry run python $runner.FullName
