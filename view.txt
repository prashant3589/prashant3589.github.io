$manualRecord = $response.records | Where-Object { $_.name -like "*Manual*" }

$logId = $manualRecord.log.id

Write-Host "Log ID: $logId"

$project = "$(System.TeamProject)"
$buildId = "$(Build.BuildId)"
$logUrl = "$org$project/_apis/build/builds/$buildId/logs/$logId?api-version=7.0"

    Authorization = "Bearer $(System.AccessToken)"}
