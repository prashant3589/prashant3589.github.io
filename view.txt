param(
[string]$EfProjectPath = "C:\EFProject",
[string]$FlywayPath = "C:\Flyway\sql",
[string]$StartVersion = 1
)

Write-Host "Starting EF → Flyway conversion..."

# Step 1: Get all EF migrations (excluding snapshot)

$migrations = Get-ChildItem "$EfProjectPath\Migrations" -Filter "*.cs" |
Where-Object { $_.Name -notlike "*ModelSnapshot*" } |
Sort-Object Name

if (-not $migrations) {
Write-Host "No migrations found."
exit
}

# Step 2: Loop through migrations

$version = $StartVersion

foreach ($migration in $migrations) {

```
# Extract migration name (remove timestamp)
$nameParts = $migration.BaseName -split "_", 2
if ($nameParts.Count -lt 2) {
    Write-Host "Skipping invalid migration: $($migration.Name)"
    continue
}

$migrationName = $nameParts[1]

# Format Flyway filename
$versionStr = ("{0:D3}" -f $version)
$flywayFile = "V${versionStr}__${migrationName}.sql"
$outputPath = Join-Path $FlywayPath $flywayFile

Write-Host "Processing migration: $migrationName → $flywayFile"

# Step 3: Generate SQL script for this migration
$tempSql = Join-Path $env:TEMP "$migrationName.sql"

Push-Location $EfProjectPath

dotnet ef migrations script `
    --idempotent `
    -o $tempSql | Out-Null

Pop-Location

# Step 4: Clean SQL (remove EF noise if needed)
$sqlContent = Get-Content $tempSql -Raw

# Optional cleanup (basic)
$sqlContent = $sqlContent -replace "^\s*Build started.*$", ""
$sqlContent = $sqlContent -replace "^\s*Build succeeded.*$", ""

# Step 5: Save to Flyway folder
Set-Content -Path $outputPath -Value $sqlContent

Write-Host "Created: $outputPath"

# Cleanup temp file
Remove-Item $tempSql -ErrorAction SilentlyContinue

$version++
```

}

Write-Host "Conversion completed successfully!"
