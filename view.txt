# Convert ServiceNow dates to DateTime (UTC)
$plannedStart = [DateTime]::Parse($planned_start_date).ToUniversalTime()
$plannedEnd   = [DateTime]::Parse($planned_end_date).ToUniversalTime()
$currentTime  = (Get-Date).ToUniversalTime()

# Allowed states
$allowedStates = @("Scheduled", "Implement")

# ---- Date Validation ----
if ($currentTime -lt $plannedStart -or $currentTime -gt $plannedEnd) {
    Write-Error "❌ Validation failed: Current time ($currentTime) is outside the approved change window ($plannedStart - $plannedEnd)."
    exit 1
}

# ---- State Validation ----
if ($allowedStates -notcontains $state) {
    Write-Error "❌ Validation failed: Change state '$state' is not allowed. Allowed states: Scheduled, Implement."
    exit 1
}

Write-Host "✅ Change validation passed."
Write-Host "State        : $state"
Write-Host "Start (UTC)  : $plannedStart"
Write-Host "End (UTC)    : $plannedEnd"





function Convert-ToDate {
    param (
        [Parameter(Mandatory = $true)]
        [string]$RawDate
    )

    return [DateTime]::ParseExact(
        $RawDate,
        "dd/MM/yyyy",
        [System.Globalization.CultureInfo]::InvariantCulture
    )
}

function Invoke-RestApiSafe {
    param (
        [Parameter(Mandatory = $true)]
        [string]$Uri,

        [Parameter()]
        [ValidateSet("GET","POST","PUT","PATCH","DELETE")]
        [string]$Method = "GET",

        [Parameter()]
        [hashtable]$Headers = @{}, 

        [Parameter()]
        [object]$Body = $null,

        [Parameter()]
        [int]$TimeoutSec = 60
    )

    try {
        Write-Host "➡️ Calling API: $Method $Uri"

        $params = @{
            Uri         = $Uri
            Method      = $Method
            Headers     = $Headers
            TimeoutSec  = $TimeoutSec
            ErrorAction = "Stop"
        }

        if ($Body) {
            $params.Body        = ($Body | ConvertTo-Json -Depth 10)
            $params.ContentType = "application/json"
        }

        $response = Invoke-RestMethod @params
        return $response
    }
    catch {
        Write-Error "❌ REST API call failed"

        if ($_.Exception.Response) {
            $statusCode = $_.Exception.Response.StatusCode.value__
            $statusDesc = $_.Exception.Response.StatusDescription

            Write-Error "HTTP Status: $statusCode $statusDesc"

            try {
                $reader = New-Object System.IO.StreamReader($_.Exception.Response.GetResponseStream())
                $errorBody = $reader.ReadToEnd()
                Write-Error "Response Body: $errorBody"
            } catch {}
        }
        else {
            Write-Error $_.Exception.Message
        }

        throw  # fail the pipeline
    }
}
